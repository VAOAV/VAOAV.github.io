操作系统之生产者消费者

Posted on 2018-05-16

生产者消费者问题（英语：Producer-consumer problem），也称有限缓冲问题（英语：Bounded-buffer problem），是一个多线程同步问题的经典案例。该问题描述了两个共享固定大小缓冲区的线程——即所谓的“生产者”和“消费者”——在实际运行时会发生的问题。生产者的主要作用是生成一定量的数据放到缓冲区中，然后重复此过程。与此同时，消费者也在缓冲区消耗这些数据。该问题的关键就是要保证生产者不会在缓冲区满时加入数据，消费者也不会在缓冲区中空时消耗数据。

模拟生产者消费者问题有两个过程，生产者将产品放入缓存区；消费者从缓存区取出产品。用循环链表模拟缓存区，随机生产价值在0~100之间的产品知道缓存区存满，消费者依次取出产品直到缓存区为空。 

#include<iostream>
#include<stdio.h>
#include<malloc.h>
using namespace std;

typedef int qelemtype;
#define ok 1
#define error 0
int maxsize;

typedef struct {
    qelemtype *base;
    int front;
    int rear;
}sqqueue;

int initqueue(sqqueue &q)
{
    q.base = (qelemtype *)malloc(maxsize * sizeof(qelemtype));
    if (!q.base) exit(0);
    q.front = q.rear = 0;
    return ok;
}
//creat box

int inqueue(sqqueue &q, qelemtype inner)
{
    q.base[q.rear] = inner;
    q.rear = (q.rear + 1) % maxsize;
    return ok;
}

int outqueue(sqqueue &q)
{
    if (q.rear == q.front) return(error);
    qelemtype inner;
    inner = q.base[q.front];
    q.front = (q.front + 1) % maxsize;
    return inner;
}

int producer(sqqueue &q)
{
    while ((q.rear + 1) % maxsize != q.front)
    {
        int goods = rand() % 100;
        printf("%3d", goods);
        inqueue(q, goods);
    }
    return ok;
}

int consumer(sqqueue &q)
{
    while ((q.front - q.rear + maxsize) % maxsize)
    {
        printf("%3d", outqueue(q));
    }
    return ok;
}

int main()
{
    scanf_s("%d", &maxsize);
    sqqueue box;
    initqueue(box);
    producer(box);
    consumer(box);
    return 0;
}
